Auth Logic: js/auth.js // Session Check & Redirect auth.onAuthStateChanged(user => { if (!user && !location.href.includes('join.html')) { location.href = 'join.html'; } else if (user && location.href.includes('join.html')) { location.href = 'index.html'; } if (user) { document.getElementById('username')?.innerText = user.displayName || user.email; } }); // Google Login function googleLogin() { const provider = new firebase.auth.GoogleAuthProvider(); provider.addScope('profile'); auth.signInWithPopup(provider); } // Join / Register handler async function handleJoin(e) { e.preventDefault(); const name = document.getElementById('name').value; const dob = document.getElementById('dob').value; const file = document.getElementById('avatar').files[0]; const user = await auth.createUserWithEmailAndPassword(`${dob}@bytebird.me`, 'defaultPass'); await user.user.updateProfile({ displayName: name }); if (file) { const ref = firebase.storage().ref(`avatars/${user.user.uid}`); await ref.put(file); const url = await ref.getDownloadURL(); await user.user.updateProfile({ photoURL: url }); } } 4. Posts Logic: js/posts.js // Fetch & render posts async function loadPosts() { const snap = await db.collection('posts').orderBy('createdAt', 'desc').get(); const container = document.getElementById('posts'); snap.forEach(doc => { const data = doc.data(); const el = document.createElement('div'); el.className = 'bg-white p-4 rounded mb-4'; el.innerHTML = `
${data.name}

${data.text}

`; container.appendChild(el); }); } document.addEventListener('DOMContentLoaded', loadPosts); // Create Post function showCreatePost() { const text = prompt('আপনার পোস্ট লিখুন:'); if (!text) return; const user = auth.currentUser; db.collection('posts').add({ uid: user.uid, name: user.displayName, avatar: user.photoURL || '', text, createdAt: firebase.firestore.FieldValue.serverTimestamp() }).then(() => location.reload()); } 5.
